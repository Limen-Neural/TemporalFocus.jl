Types

SpikeEvent

struct SpikeEvent
    neuron_id::Int
    t::Float32
    value::Float32
end

SpikeEvent(neuron_id::Integer, t::Real, value::Real = 1.0f0)

A single spike: neuron index, time, and optional amplitude (default 1.0f0).

SpikeTrain

struct SpikeTrain
    events::Vector{SpikeEvent}
end

SpikeTrain()
SpikeTrain(events::AbstractVector{<:SpikeEvent})

Ordered collection of spike events used by discrete and temporal attention kernels.

TemporalBuffer

struct TemporalBuffer
    window::Float32
    events::Vector{SpikeEvent}
end

TemporalBuffer(window::Real, events::AbstractVector{<:SpikeEvent} = SpikeEvent[])

Sliding window of recent spikes for continuous attention.

Buffer management

prune!

prune!(buffer::TemporalBuffer, current_time::Real) -> TemporalBuffer

Remove events older than buffer.window relative to current_time. Mutates buffer in place.

Temporal weighting

temporal_weight

temporal_weight(dt::Real, τ::Real) -> Float32

Exponential recency weight $e^{-|dt|/τ}$. Requires τ > 0.

Attention kernels

spike_attention_discrete

spike_attention_discrete(
    source_spikes::SpikeTrain,
    context_spikes::SpikeTrain,
    readout::AbstractMatrix,
)

Coincidence attention: accumulate products of co-located source/context spike values, then apply readout (transpose(readout) * attention).

spike_attention_temporal

spike_attention_temporal(
    source_spikes::SpikeTrain,
    context_spikes::SpikeTrain,
    readout::AbstractMatrix;
    τ::Real = 1.0f0,
)

Like discrete attention, but each pair is scaled by temporal_weight(Δt, τ).

spike_attention_continuous

spike_attention_continuous(
    source_buffer::TemporalBuffer,
    context_buffer::TemporalBuffer,
    readout::AbstractMatrix;
    τ::Real = 1.0f0,
)

Continuous-time attention over buffered events within the shared window, decayed by temporal_weight and projected through readout.

Normalization

normalize_l1!(weights::AbstractVector{<:AbstractFloat}) -> weights

normalize_max!

normalize_max!(weights::AbstractVector{<:AbstractFloat}) -> weights

In-place max normalization. No-op when the peak is zero.
